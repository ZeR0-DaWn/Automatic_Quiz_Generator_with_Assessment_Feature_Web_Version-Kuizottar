  <!-- ðŸŒ« Dark Fog Effect -->
  <script>
    const canvas = document.getElementById("fogCanvas");
    const ctx = canvas.getContext("2d");
    let w, h, t = 0;
    function resize() {
      w = canvas.width = window.innerWidth;
      h = canvas.height = window.innerHeight;
    }
    window.addEventListener("resize", resize);
    resize();

    // Simple smooth noise function
    function noise(x, y) {
      return (Math.sin(x * 0.02) + Math.sin(y * 0.02) + Math.sin((x + y + t * 10) * 0.01)) * 0.5 + 0.5;
    }

    function drawFog() {
      const img = ctx.createImageData(w, h);
      const d = img.data;
      for (let y = 0; y < h; y += 2) {
        for (let x = 0; x < w; x += 2) {
          const n = noise(x, y);
          const i = (y * w + x) * 4;
          const intensity = Math.floor(n * 60);
          d[i] = intensity * 0.6;
          d[i + 1] = intensity * 0.4;
          d[i + 2] = intensity * 0.9;
          d[i + 3] = 60; // transparency
        }
      }
      ctx.putImageData(img, 0, 0);
      t += 0.002;
      requestAnimationFrame(drawFog);
    }
    drawFog();
  </script>
